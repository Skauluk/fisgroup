Select "Авторы"."ФИО", count("Истории Использования"."Дата выдачи") from "Авторы"
join "Связь Книги-Авторы" on "Авторы"."Код автора" = "Связь Книги-Авторы"."Код автора"
join "Книги" on "Связь Книги-Авторы"."ISBN" = "Книги".ISBN
join "Экземпляры" on "Книги"."ISBN" = "Экземпляры"."ISBN"
join "Истории Использования" on "Экземпляры"."Инвентарный Номер" = "Истории Использования"."Инвентарный Номер"
where "Истории Использования" > date(current_date, "-1 year")
group by "Авторы"."ФИО"
order by count("Истории Использования") desc
limit 1
